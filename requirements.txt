# Instrucciones para iniciar el proyecto

## Inicializar proyecto
npm init -y

## Instalar dependencias
npm install express@4.x.x \
             bcryptjs@2.x.x \
             jsonwebtoken@8.x.x \
             helmet@4.x.x \
             csurf@1.x.x \
             cookie-parser@1.x.x \
             express-rate-limit@5.x.x \
             pg@8.x.x \
             dotenv@10.x.x
